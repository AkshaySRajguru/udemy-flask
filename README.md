## Run Flask Application
# Bash
$ export FLASK_APP=flaskr

$ flask run

> set FLASK_APP=flaskr
> set FLASK_ENV=development
> flask run
# Powershell
> $env:FLASK_APP = "flaskr"
> $env:FLASK_ENV = "development"
> flask run
