## pip install Flask
### Python 3.9.6

# Run Flask Application
## Bash
 export FLASK_APP=flaskr <br />
 export FLASK_ENV=development <br />
 flask run <br />
## CMD
 set FLASK_APP=flaskr <br />
 set FLASK_ENV=development <br />
 flask run <br />
## Powershell
 $env:FLASK_APP = "flaskr" <br />
 $env:FLASK_ENV = "development" <br />
 flask run <br />
